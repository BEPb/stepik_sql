```sql 
CREATE TABLE book(                              /*Создать таблицу с именем book:*/
    book_id INT PRIMARY KEY AUTO_INCREMENT,     /*Столбец book_id целое число атоматическое заполнение:*/
    title VARCHAR(50),
    author VARCHAR(30),
    price DECIMAL(8, 2),
    amount INT);
```
